210static void ExpiracionFuerte(float presion_neta_kPa_1){
211
212 /* 1. Calculo cuántos leds prender utilizando regla de tres (mapeo lineal) */
213 /* Rango de presión: 0.84kPa (0 LEDs) a 2.7kPa (3.54-0.84) (12 LEDs) */
214
215 int leds_a_encender = (int)((presion_neta_kPa_1 / 2.69f) * 12.0f);
216
217 /* Contol de límites (si el paciente sopla con demás fuerza la regla de tres puede dar > 12LEDs) */
218 if (leds_a_encender < 0) leds_a_encender = 0;
219 if (leds_a_encender > 12) leds_a_encender = 12;
220
221 /* 2. Bucle para recorrer los 12 LEDs */
222 for (int i = 0; i < 12; i++) {
223
224 if (i < leds_a_encender) {
225
226 /* Indico el color del LED encendido según su posición */
227 if (i<3) {
228 pixel_data[i] = NEOPIXEL_COLOR_RED; /* LEDs 1, 2 ,3 --> ROJOS */
229 } else if (i<6) {
230 pixel_data[i] = NEOPIXEL_COLOR_YELLOW; /* LEDs 4, 5, 6 --> AMARILLOS */
231 } else if (i<9) {
232 pixel_data[i] = NEOPIXEL_COLOR_BLUE; /* LEDs 7, 8, 9 --> AZULES */
233 } else if (i<12) {
234 pixel_data[i] = NEOPIXEL_COLOR_VIOLET; /* LEDs 10, 11, 12 --> VIOLETAS */
235 }
236
237 } else {
238 pixel_data[i] = NEOPIXEL_COLOR_BLACK; /* LEDs restantes --> APAGADOS */
239 }
240
241 }
242
243 NeoPixelSetArray(pixel_data);
244
245 /* 3. Análisis del soplo */
246 if (presion_neta_kPa_1 > 0.0f) {
247 // EL PACIENTE ESTÁ SOPLANDO
248 soplando = true;
249
250 // Cronómetro: sumamos los 50ms de este ciclo
251 tiempo_espiracion_ms += periodo_muestreo;
252
253 // Integración (Área bajo la curva): sumamos Flujo * Tiempo(en segundos)
254 volumen_relativo_acumulado += (presion_neta_kPa_1 * (periodo_muestreo / 1000.0f));
255
256 // Récord de presión: ¿es mayor al máximo de este soplido?
257 if (presion_neta_kPa_1 > presion_maxima) {
258 presion_maxima = presion_neta_kPa_1;
259 }
260
261 } else if (soplando == true && presion_neta_kPa_1 == 0.0f) {
262 // EL PACIENTE ACABA DE TERMINAR EL SOPLIDO (Flanco de bajada)
263 float tiempo_soplado_segundos = tiempo_espiracion_ms / 1000.0f;
264
265 /* 4. MOSTRAMOS LOS RESULTADOS */
266 snprintf(msg1, sizeof(msg1), "*KMax:%.2f kPa | Dur:%.1f s | VolumenRelatvo:%.2f\n*",
267 presion_maxima, tiempo_soplado_segundos, volumen_relativo_acumulado);
268
269 if (BleStatus() == BLE_CONNECTED) {
270 BleSendString(msg1);
271 }
272
273 // RESETEAMOS TODO para el siguiente soplido
274 soplando = false;
275 tiempo_espiracion_ms = 0;
276 volumen_relativo_acumulado = 0.0f;
277 presion_maxima = 0.0f;
278 }

288static void ExpiracionMantenida(float presion_neta_kPa_2){
289
290 bool enviar_mensaje = false;
291
292 // Si el temporizador visual está corriendo, mantenemos el color final y descontamos tiempo
293 if (tiempo_espera_visual_ms > 0) {
294 if (tiempo_espera_visual_ms >= periodo_muestreo) {
295 tiempo_espera_visual_ms -= periodo_muestreo;
296 } else {
297 tiempo_espera_visual_ms = 0;
298 }
299 return; // Salimos de la función para mantener el color en el NeoPixel sin recalcular
300 }
301
302 /* El paciente está espirando?*/
303 if (presion_neta_kPa_2 > 0.0f) {
304 soplando = true;
305 } else soplando = false;
306
307 /* Controlo el neopixel*/
308 if(soplando == true && tiempo_espiracion_ms < 2000){
309 /* Estado: ARRANCA LA CUENTA REGRESIVA */
310 tiempo_espiracion_ms += periodo_muestreo; /* Cronómetro: le sumo los 50ms de este ciclo */
311
312 /*Calculo cuantos leds apagar por segundo (3 cada 1000ms)*/
313 int leds_a_apagar = (tiempo_espiracion_ms *12) / 2000;
314 int leds_encendidos = 12 - leds_a_apagar;
315
316 for (int i = 0; i < 12; i++){
317 if (i < leds_encendidos) {
318 pixel_data[i] = NEOPIXEL_COLOR_ROSE; /* Le aviso que está sensando el soplo */
319 } else {
320 pixel_data[i] = NEOPIXEL_COLOR_BLACK; /* Se van apagando */
321 }
322 }
323 NeoPixelSetArray(pixel_data);
324
325 } else if (soplando == true && tiempo_espiracion_ms >= 2000) {
326 /* Estado: TIEMPO COMPLETADO. El paciente sigue espirando*/
327 /*Se ponen en verde*/
328 for (int i = 0; i < 12; i++) {
329 pixel_data[i] = NEOPIXEL_COLOR_GREEN;
330 }
331 NeoPixelSetArray(pixel_data);
332
333 } else if (soplando == false && tiempo_espiracion_ms >= 2000) {
334 /* Estado: No espira porque TERMINÓ el soplo exitosamente*/
335 /* Escribo mensaje */
336 snprintf(msg2, sizeof(msg2), "*MEXITO!\n*");
337 enviar_mensaje = true;
338
339 // En lugar de vTaskDelay, usamos el temporizador pasivo
340 tiempo_espera_visual_ms = 1500;
341 tiempo_espiracion_ms = 0;
342 soplando = false;
343
344 } else if (soplando == false && tiempo_espiracion_ms > 0 && tiempo_espiracion_ms < 2000) { /* No espira porque CORTÓ ANTES el aire*/
345 /* Escribo mensaje */
346 snprintf(msg2, sizeof(msg2), "*MFALLO! duracion: %.1fs\n*", tiempo_espiracion_ms / 1000.0f);
347 enviar_mensaje = true;
348
349 /*Se ponen en rojo*/
350 for (int i = 0; i < 12; i++) {
351 pixel_data[i] = NEOPIXEL_COLOR_RED;
352 }
353 NeoPixelSetArray(pixel_data);
354
355 // En lugar de vTaskDelay, usamos el temporizador pasivo
356 tiempo_espera_visual_ms = 1500;
357 tiempo_espiracion_ms = 0;
358 soplando = false;
359
360 } else {
361 /* Estado: EN ESPERA. Aún no comenzó el soplo y el tiempo = 0*/
362 for (int i = 0; i < 12; i++) {
363 pixel_data[i] = NEOPIXEL_COLOR_WHITE; /* Para indicar que está listo para comenzar */
364 }
365 NeoPixelSetArray(pixel_data);
366 }
367
368 /* Envío del mensaje */
369 if (enviar_mensaje==true){
370 if(BleStatus() == BLE_CONNECTED) {
371 BleSendString(msg2);
372 }
373 }

383static void Sensado(void *pvParameters){
384
385 while (true) {
386
387 if (medir_presion==true) {
388
389 float presion_medida = MP3v5050ReadPressure_kPa();
390
391 /* 1. Resto la linea base (taramos). Sin soplar sensa 0.86kPa*/
392 float presion_neta = presion_medida - 0.86f; // restamos el reposo
393 if (presion_neta < 0.15f) {
394 presion_neta = 0.0f; // Evitamos valores negativos si el sensor fluctúa a 0.83
395 }
396
397 /* 2. Control del NeoPixel según el tipo de expiración */
398 if (expiracion_fuerte == true){
399 ExpiracionFuerte(presion_neta);
400 //Imprimir(NULL);
401 }
402 else if (expiracion_mantenida == true){
403 ExpiracionMantenida(presion_neta);
404 }
405
406 } else {
407 // Limpiamos el arreglo a mano
408 for(int i = 0; i < 12; i++){
409 pixel_data[i] = NEOPIXEL_COLOR_BLACK; /* Apagamos todos los LEDs */
410 }
411 NeoPixelSetArray(pixel_data);
412 }
413
414 vTaskDelay(periodo_muestreo / portTICK_PERIOD_MS);
415 }

438void app_main(void){

449 SwitchActivInt(SWITCH_2, Tecla2_interrupcion, NULL);
450
451 /* Inicializacion de bluetooth*/
452 ble_config_t ble_config = {
453 .device_name = "ESP_EDU_Naza",
454 .func_p = RecibirDatoBluetooth
455 };
456
457 BleInit(&ble_config);
458
459 /* Inicalización de los uart */
460 serial_config_t configuracion_uart = {
461 .port = UART_PC,
462 .baud_rate = 115200,
463 .func_p = NULL,
464 .param_p = NULL
465 };
466
467 UartInit(&configuracion_uart);
468
469 xTaskCreate(&Sensado, "Sensado", 2048, NULL, 5, &Sensado_task_handle);
470 xTaskCreate(&ControlBle, "Control del bluetooth", 2048, NULL, 5, &ControlBle_task_handle);

472}
